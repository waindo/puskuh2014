<script>
var myFolders;
function doOnLoad() {
    myFolders = new dhtmlxFolders("folders_container");
    myFolders.setItemType("ficon");
    myFolders.setUserData("icons_src_dir", "../common/images");
    myFolders.loadXML("../common/files_ext.xml", "../../codebase/types/ficon.xsl");
}
function changeType(type) {
    myFolders.setItemType(type, "../../codebase/types/" + type + ".xsl");
}
//filter items by attribute "name";
function filterItemsByName(obj, mask) {
    var aName = obj.data.dataObj.getAttribute("name");
    return (aName.toString().toLowerCase().indexOf(mask.toString().toLowerCase()) == 0 || mask == "");
}
//filter items by value of subtag "filesize";
function filterItemsBySizeGT(obj, mask) {
    try {
        var objSize = parseInt(obj.data.dataObj.getElementsByTagName("filesize")[0].firstChild.nodeValue);
    } catch(e) {
        var objSize = 0;
    }
    return (objSize > mask);
}
</script>
